-- VIEW COM AS MAIORES RECEITAS POR CLIENTE LIMITANDO POR TRÊS E ORDENANDO PELAS MAIORES RECEITAS

CREATE VIEW MAIORES_RECEITAS_POR_CLIENTE AS
SELECT CLIENTE.NOME AS CLIENTE,
      (SELECT SUM(ENCOMENDA.VALOR)
	       FROM ENCOMENDA
        WHERE ENCOMENDA.CLIENTE_CODIGO = CLIENTE.CODIGO) AS RECEITA
  FROM CLIENTE
 ORDER BY RECEITA DESC
 LIMIT 3;

-- VIEW DE ARRECADAÇÃO POR PLACA ORDENANDO APENAS OS CODIGOS DA PLACA DE FORMA CRRESCENTE

CREATE VIEW ARRECADACAO_POR_PLACA AS
SELECT PLACA.CODIGO AS CODIGO_PLACA,
       SUM(ENCOMENDA.VALOR) AS ARRECADACAO
  FROM PLACA
  JOIN ITEM
    ON ITEM.PLACA_CODIGO = PLACA.CODIGO
  JOIN ENCOMENDA
    ON ITEM.ENCOMENDA_CODIGO = ENCOMENDA.CODIGO
 GROUP BY PLACA.CODIGO
 ORDER BY PLACA.CODIGO ASC
