-- CONSULTA CLIENTE COMPLETO

SELECT CLIENTE.NOME,
       CLIENTE.CPF,
	   CONTATO.TIPO,
	   CONTATO.INFORMACAO
  FROM CLIENTE
  JOIN CONTATO
    ON CLIENTE.CODIGO = CONTATO.CLIENTE_CODIGO

-- CONSULTA DE ITEM EXIBINDO DADOS PRINCIPAIS

SELECT ITEM.CODIGO AS ITEM,
       ITEM.FRASE AS FRASE,
       ENCOMENDA.SITUACAO_ENCOMENDA AS SITUACAO_ENCOMENDA,
       ENCOMENDA.VALOR AS VALOR,
       MATERIAL.DESCRICAO AS MATERIAL,
       CLIENTE.NOME AS CLIENTE
	   
  FROM ITEM
  JOIN ENCOMENDA
    ON ENCOMENDA.CODIGO = ITEM.ENCOMENDA_CODIGO
  JOIN CLIENTE
    ON ENCOMENDA.CLIENTE_CODIGO = CLIENTE.CODIGO
  JOIN PLACA
    ON PLACA.CODIGO = ITEM.PLACA_CODIGO
  JOIN MATERIAL
    ON PLACA.MATERIAL_CODIGO = MATERIAL.CODIGO
 WHERE ITEM.ENCOMENDA_CODIGO = 3
   AND PLACA.MATERIAL_CODIGO = 4

-- CONSULTA PARA SABER QUANTIDADE DE ENCOMENDAS POR CLIENTE

SELECT CLIENTE.NOME AS cLIENTE,
       COUNT(*) AS QUANTIDADE_DE_ENCOMENDAS
  FROM ENCOMENDA
  JOIN CLIENTE
    ON CLIENTE.CODIGO = ENCOMENDA.CLIENTE_CODIGO
 GROUP BY CLIENTE.NOME

-- CONSULTA PARA SABER A QUANTIDADE DE ENCOMENDAS POR CLIENTE POR DATA (MENOR QUE A DATA X)

SELECT CLIENTE.NOME AS cLIENTE,
       COUNT(*) AS QUANTIDADE_DE_ENCOMENDAS
  FROM ENCOMENDA
  JOIN CLIENTE
    ON CLIENTE.CODIGO = ENCOMENDA.CLIENTE_CODIGO
 WHERE DATA < '16/05/2026'
 GROUP BY CLIENTE.NOME

 -- CONSULTA PARA SABER A QUANRTIDADE DE ENCOMENDAS POR CLIENTE MANTENDO OS AGRUPAMENTOS QUE SÃO IGUAIS A QUATRO REGISTROS (4 POR CONTA DE TER SIDO CADASTRADO 4 ENCOMENDAS PARA CADA CLIENTE)

SELECT CLIENTE.NOME AS CLIENTE,
       COUNT(*) AS QUANTIDADE_DE_ENCOMENDAS
  FROM ENCOMENDA
  JOIN CLIENTE 
    ON CLIENTE.CODIGO = ENCOMENDA.CLIENTE_CODIGO
 GROUP BY CLIENTE.NOME
HAVING COUNT(*) = 4

-- CONSULTA PARA SABER A QUANNTIDADE DE ENCOMENDAS POR CLIENTE FILTRANDO POR UMA DATA MENOR DO QUE A INFORMADA E MANTENDO APENAS OS AGRUPAMENTOS MAIORES QUE 2 REGISTROS

SELECT CLIENTE.NOME AS cLIENTE,
       COUNT(*) AS QUANTIDADE_DE_ENCOMENDAS
  FROM ENCOMENDA
  JOIN CLIENTE
    ON CLIENTE.CODIGO = ENCOMENDA.CLIENTE_CODIGO
 WHERE DATA < '16/05/2026'
 GROUP BY CLIENTE.NOME
HAVING COUNT(*) > 2
