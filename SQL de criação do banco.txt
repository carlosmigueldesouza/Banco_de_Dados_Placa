  CELULAR
  EMAIL
}
table cliente{
  codigo serial [unique, increment, pk]
  cpf char(11) [unique, not null]
  nome varchar(100)
  }

  table contato{
    codigo serial [unique, pk, increment]
    cliente_codigo integer [not null]
    tipo tipo_contato [not null]
    informacao varchar(255) [not null]
  }

  table cor{
    codigo serial [unique, pk, increment]
    descricao varchar(100) [not null]
    fundo boolean [not null]
    frase boolean [not null]
  }

  table encomenda{
    codigo serial [unique, pk, increment]
    data timestamp [not null]
    entrega timestamp [not null]
    cliente_codigo integer [not null]
    desconto_codigo integer
  }

  table item{
    codigo serial [unique, pk, increment]
    encomenda_codigo integer [not null]
    placa_codigo integer [not null]
    frase varchar(50) [not null]
    cor_fundo_codigo integer [not null]
    cor_frase_codigo integer [not null]
  }

  table desconto{
    codigo serial [unique, pk, increment]
    valor_minimo decimal(10,2) [not null]
    valor_maximo decimal(10,2) [not null]
    percentual decimal(5,2) [not null]
  }

  table placa{
    codigo serial [unique, pk, increment]
    largura integer [not null]
    altura integer [not null]
    material_codigo integer [not null]
  }

  table material{
    codigo serial [unique, not null, increment]
    descricao varchar(20) [not null]
    valor_letra decimal(10,2) [not null]
    custo decimal(10,2) [not null]
  }
