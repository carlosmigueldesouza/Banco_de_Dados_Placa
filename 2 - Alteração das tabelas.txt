-- Criar Tabela RECEBIMENTO
CREATE TABLE "RECEBIMENTO" (
	"CODIGO"           SERIAL PRIMARY KEY,
	"ENCOMENDA_CODIGO" INTEGER       NOT NULL,
	"VALOR_SINAL"      NUMERIC(10,2) NOT NULL,
	"VALOR_RESTANTE"   NUMERIC(10,2) NOT NULL,

	CONSTRAINT FK_RECEBIMENTO_ENCOMENDA
	 FOREIGN KEY (ENCOMENDA_CODIGO)
	 REFERENCES ENCOMENDA(CODIGO)
);
